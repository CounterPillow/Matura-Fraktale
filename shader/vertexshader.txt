#extension GL_ARB_gpu_shader_fp64 : enable
attribute vec2 position;

varying dvec2 txcoords;

uniform double zoom;
uniform dvec2 offset;
uniform double ratio;

void main()
{
	gl_Position = vec4(position, 0.0, 1.0);
	txcoords = dvec2(position.x, position.y) * (dvec2(0.5) * zoom) + dvec2(0.5) + offset;
	txcoords.y *= double(ratio);
}
